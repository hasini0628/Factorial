import java.util.Scanner; class Factorial { // method to calculate factorial using recursion long calculate(int n) { if (n == 0 || n == 1) return 1; else return n * calculate(n - 1); } } public class Main { public static void main(String[] args) { Scanner sc = new Scanner(System.in); System.out.print("Enter a number: "); int n = sc.nextInt(); Factorial obj = new Factorial(); long result = obj.calculate(n); System.out.println("Factorial of " + n + " = " + result); } }
